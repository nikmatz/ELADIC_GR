  {Maxima: Παράγωγος — Κανόνες, Ανώτερης Τάξης, Εφαπτόμενη}%
  {%
    \label{cas:ch8_derivatives}
    \begin{enumerate}[label=(\alph*),itemsep=2pt]
      \item Υπολογίστε με \texttt{diff} τις παραγώγους:
        $f(x)=3x^4-5x^2+2x-7$, $\sin x$, $e^{3x}$, $\ln(x^2+1)$.
        Επαληθεύστε εφαρμόζοντας τους κανόνες χειρωνακτικά.
      \item Εφαρμόστε κανόνα γινομένου, πηλίκου και αλυσίδας:
        $(x^2\sin x)'$, $\;\bigl(\frac{x^3+1}{x^2-1}\bigr)'$,
        $\;[\sin(x^2+1)]'$.
      \item Για $f(x)=e^x\cos x$ υπολογίστε $f'$, $f''$, $f'''$,
        $f^{(4)}$ και επαληθεύστε $f^{(4)}=-4f$.
      \item Υπολογίστε την εφαπτόμενη ευθεία του
        $f(x)=x^3-2x+1$ στο $x_0=1$.
      \item Με τον \textbf{ορισμό}: $f'(x)=\lim_{h\to0}\frac{f(x+h)-f(x)}{h}$,
        υπολογίστε \texttt{diff} μέσω \texttt{limit} για $x^2$ και $\sqrt{x}$.
    \end{enumerate}
    \smallskip
    \textbf{Εντολές Maxima:}
    \texttt{diff(f,x)}, \texttt{diff(f,x,n)},
    \texttt{at(diff(f,x),x=a)}, \texttt{trigsimp},
    \texttt{limit((f(x+h)-f(x))/h,h,0)}

\subsection*{Δραστηριότητα Python}

\begin{pybox}{Python: Παράγωγος — SymPy, NumPy, Γραφικά}%
             {https://nikmatz.github.io/ELADIC_GR/code/ch08/}%
             {qr_21ad25b531c5.png}
\label{py:ch8_derivatives}
\begin{enumerate}[label=(\alph*),itemsep=2pt]
  \item Με \texttt{sympy.diff} υπολογίστε παραγώγους για τις
    συναρτήσεις της δραστηριότητας Maxima. Συγκρίνετε.
  \item Εφαρμόστε \texttt{lambdify} για να μετατρέψετε τις
    συμβολικές παραγώγους σε αριθμητικές. Σχεδιάστε $f$, $f'$,
    $f''$ για $f(x)=x^3-2x+1$ στο $[-2,\,2.5]$.
  \item Υπολογίστε αριθμητικά $f'(x) \approx \frac{f(x+h)-f(x)}{h}$
    για $f(x)=\sin x$ με \texttt{np.gradient}. Συγκρίνετε με $\cos x$.
  \item Σχεδιάστε την εφαπτόμενη ευθεία στο $x_0=1$ και
    δείξτε τη γεωμετρικά (Matplotlib).
  \item \textit{Προαιρετικό:} Λογαριθμική παραγώγιση:
    επαληθεύστε $(x^x)' = x^x(\ln x+1)$.
\end{enumerate}
\medskip
\textbf{Βιβλιοθήκες / Εντολές:}
\texttt{sympy.diff()}, \texttt{sympy.lambdify()},
\texttt{np.gradient()}, \texttt{f.subs(x,a)},
\texttt{matplotlib.pyplot}
\end{pybox}
