\casactivbox%
  {Maxima: Όρια — Άρση Αοριστίας, L'Hôpital, Taylor}%
  {%
    \label{cas:ch7_limits}
    \begin{enumerate}[label=(\alph*),itemsep=2pt]
      \item Υπολογίστε με \texttt{limit} τα παρακάτω όρια και
        επαληθεύστε τα αποτελέσματα χειρωνακτικά:
        \[
          \lim_{x\to 3}\frac{x^2-9}{x-3}, \quad
          \lim_{x\to 0}\frac{\sin x}{x}, \quad
          \lim_{x\to 0}\frac{e^x-1}{x}, \quad
          \lim_{x\to+\infty}\frac{3x^2-2x+1}{x^2+5}.
        \]
      \item Ελέγξτε ύπαρξη ορίου στο 0 για $f(x)=|x|/x$:
        υπολογίστε \texttt{limit(f,x,0,plus)} και
        \texttt{limit(f,x,0,minus)}.
      \item Για $f(x)=(x^2-4)/(x-2)$ με $f(2)=4$: επαληθεύστε
        ότι $\lim_{x\to 2}f(x)=f(2)$ και χαρακτηρίστε τη
        συνέχεια.
      \item Αναπτύξτε σε σειρά Maclaurin (\texttt{taylor}) τις
        $\sin x$, $e^x$, $\ln(1+x)$ έως 5η τάξη.
    \end{enumerate}
    \smallskip
    \textbf{Εντολές Maxima:}
    \texttt{limit(f,x,a)}, \texttt{limit(f,x,a,plus/minus)},
    \texttt{limit(f,x,inf)}, \texttt{taylor(f,x,0,n)}
  }%
  {https://nikmatz.github.io/ELADIC_GR/code/ch07/}%

\subsection*{Δραστηριότητα Python}

\begin{pybox}{Python: Όρια, Συνέχεια, Taylor — SymPy \& Matplotlib}%
             {https://nikmatz.github.io/ELADIC_GR/code/ch07/}%
             {qr_1db7857dd739.png}
\label{py:ch7_limits}
\begin{enumerate}[label=(\alph*),itemsep=2pt]
  \item Με \texttt{sympy.limit} υπολογίστε τα όρια της
    δραστηριότητας Maxima. Συγκρίνετε αποτελέσματα.
  \item Γράψτε συνάρτηση \texttt{check\_continuity(f, a, f\_at\_a)}
    που χρησιμοποιεί \texttt{limit} για να ελέγξει συνέχεια.
    Δοκιμάστε τη στις $f$, $g(x)=\sin(x)/x$ με $g(0)=1$.
  \item Με \texttt{sympy.series} αναπτύξτε $\sin x$, $e^x$,
    $\ln(1+x)$, $\frac{1}{1-x}$ γύρω από το 0.
  \item Σχεδιάστε $\sin(x)/x$, τα πολυώνυμα Taylor τάξης
    1, 3, 5, 7 και το σφάλμα προσέγγισης στο $[-2\pi,2\pi]$
    (Matplotlib).
\end{enumerate}
\medskip
\textbf{Βιβλιοθήκες / Εντολές:}
\texttt{sympy.limit()}, \texttt{sympy.series()},
\texttt{sympy.oo}, \texttt{matplotlib.pyplot},
\texttt{np.where()} (τμηματικές συναρτήσεις)
\end{pybox}
